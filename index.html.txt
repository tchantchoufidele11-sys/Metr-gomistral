async function api(action, data = {}) {
    const res = await fetch('nexus.php', { method:'POST', headers:{'Content-Type':'application/json'}, body:JSON.stringify({action, ...data}) });
    const txt = await res.text();
    try { return JSON.parse(txt); } catch(e) { return {ok:false, error:'Réponse serveur invalide (non-JSON): ' + txt.substring(0,100)}; }
}
function toast(msg, dur=2500) { const t=document.getElementById('toast'); t.textContent=msg; t.classList.add('show'); setTimeout(()=>t.classList.remove('show'), dur); }
function escapeHtml(s) { return String(s).replace(/[&<>"']/g, c=>({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[c])); }
function renderMarkdown(text) {
    let html = escapeHtml(text);
    html = html.replace(/```(\w*)\n([\s\S]*?)```/g, (m,l,c)=>`<pre><code>${c}</code></pre>`);
    html = html.replace(/`([^`]+)`/g, '<code>$1</code>');
    html = html.replace(/\*\*([^\*]+)\*\*/g, '<strong>$1</strong>');
    html = html.replace(/\*([^\*]+)\*/g, '<em>$1</em>');
    html = html.replace(/^### (.+)$/gm, '<h3 style="font-family:Space Grotesk;margin:10px 0 6px;">$1</h3>');
    html = html.replace(/^## (.+)$/gm, '<h2 style="font-family:Space Grotesk;margin:12px 0 6px;">$1</h2>');
    html = html.replace(/^- (.+)$/gm, '• $1<br>');
    return html;
}

            }
            messagesEl.scrollTop = messagesEl.scrollHeight;
        }
        document.querySelector(`#${resultId} .bulk-inline-title`).textContent = '✓ Traitement terminé';
    }

    const body = document.getElementById('consoleBody');
    const filtered = state.logFilter === 'all' ? state.logs : state.logs.filter(l => state.logFilter === 'error' ? l.status === 'error' : l.type === state.logFilter);
    document.getElementById('logCount').textContent = `${filtered.length} entrées`;
    const header = body.querySelector('.log-row');
    body.innerHTML = '';
    body.appendChild(header);
    if (!filtered.length) { body.innerHTML += '<div class="empty-state">Aucun log</div>'; return; }
    filtered.forEach(l => {
        const row = document.createElement('div');
        row.className = 'log-row';
        row.innerHTML = `
            <div class="log-ts">${l.timestamp?.substring(11,19)||'—'}</div>
            <div class="log-type ${l.type}">${l.type}</div>
            <div class="log-action">${escapeHtml(l.action||'')}</div>
            <div class="log-model">${escapeHtml(l.source||l.model||'')}</div>
            <div class="log-ms">${l.ms||0}</div>
            <div class="log-payload" title="${escapeHtml((l.payload||'')+' | '+(l.response||''))}"><span class="log-status ${l.status}">${l.status}</span> ${escapeHtml((l.payload||'')+(l.response?' → '+l.response:''))}</div>`;
        body.appendChild(row);
    });
}

async function loadStats() {

    }
}
